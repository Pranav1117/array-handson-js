   let freq=[1001];
   
   for(let i=0;i<1001;i++) {
     freq[i]=0;
   }
   
   for(let i=0;i<N;i++) {

  11. Birthday Game

    12. Answer:

    function Birthday_Game(arr,D ,M) {
 var noOfWays=0;
 for(var i=0; i<(arr.length) ; i++){
   var j=0;
   var currSum=0;
   while(j<M){
     currSum=currSum+arr[i+j];
     j++;
   }
   if(currSum ===D)
   {
     noOfWays++
   }
 }
return noOfWays;
}
